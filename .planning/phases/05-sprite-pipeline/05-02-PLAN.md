    ```python
    # ======== CDN 可达性探针（pilot 第 0 步）========
    # INPUT:  data/apphash.json (冻结的 sekai-apphash JP production)
    #         src/data/mysekaiFixtures.json (master DB)
    # OUTPUT: .planning/phases/05-sprite-pipeline/05-02-CDN-PROBE.md
    #         + 3 个临时下载 bundle 用于 SHA256 校验
    # POS:    scripts/sprite-pipeline/pipeline/cdn_probe.py
    # 备注：sssekai (2026-02-25) 与 sekai-apphash (2026-02-06) 均已归档，
    #       本任务先确认冻结哈希仍能对当前 JP CDN 鉴权，再决定是否继续 pilot。

    import hashlib, json, os, subprocess, sys, tempfile
    from datetime import datetime, timezone
    from pathlib import Path

    APPHASH_JSON = Path(__file__).resolve().parents[1] / "data" / "apphash.json"
    FIXTURES_JSON = Path(__file__).resolve().parents[2].parent / "src" / "data" / "mysekaiFixtures.json"
    REPORT_PATH = Path(__file__).resolve().parents[2].parent / ".planning" / "phases" / "05-sprite-pipeline" / "05-02-CDN-PROBE.md"

    def _load_credentials() -> tuple[str, str]:
        d = json.loads(APPHASH_JSON.read_text())
        # 已展平：直接取顶层；否则到 raw 列表里找 production
        if "appVersion" in d and "appHash" in d:
            return d["appVersion"], d["appHash"]
        for entry in d.get("raw", []):
            if "production" in str(entry.get("environment", "")):
                return entry["appVersion"], entry["appHash"]
        raise SystemExit("apphash.json missing production credentials")

    def _pick_probe_fixtures() -> list[dict]:
        all_fx = json.loads(FIXTURES_JSON.read_text())
        outdoor = sorted(
            [fx for fx in all_fx if fx.get("mysekaiSettableSiteType") != "room"],
            key=lambda f: f["id"],
        )
        if len(outdoor) < 3:
            raise SystemExit(f"only {len(outdoor)} outdoor fixtures; need ≥3")
        return [outdoor[0], outdoor[len(outdoor)//2], outdoor[-1]]

    def _try_download(app_version: str, app_hash: str, bundle_name: str, dest: Path) -> dict:
        # sssekai 的 --download-filter 是正则，bundle_name 里的点要转义
        pattern = "^" + bundle_name.replace(".", r"\.") + "$"
        cmd = [
            "sssekai", "abcache",
            "--db", str(dest / "abcache.db"),
            "--app-region", "jp",
            "--app-version", app_version,
            "--app-appHash", app_hash,
            "--download-filter", pattern,
            "--download-ensure-deps",
            "--download-dir", str(dest / "bundles"),
        ]
        result = subprocess.run(cmd, capture_output=True, text=True, timeout=180)
        bundle_path = dest / "bundles" / f"{bundle_name}.unity3d"
        info = {
            "bundle": bundle_name,
            "exit_code": result.returncode,
            "stderr_tail": result.stderr[-400:].strip(),
        }
        if bundle_path.exists() and bundle_path.stat().st_size > 0:
            info["bytes"] = bundle_path.stat().st_size
            info["sha256"] = hashlib.sha256(bundle_path.read_bytes()).hexdigest()
            info["status"] = "OK"
        else:
            info["bytes"] = 0
            info["sha256"] = None
            info["status"] = "FAILED"
        return info

    def _verdict(results: list[dict]) -> str:
        oldest, mid, newest = results
        if all(r["status"] == "OK" for r in results):
            return "LIVE_OK"
        if oldest["status"] == "OK":
            return "PARTIAL"
        return "FAILED"

    def main() -> int:
        app_version, app_hash = _load_credentials()
        probes = _pick_probe_fixtures()
        results = []
        with tempfile.TemporaryDirectory(prefix="cdn-probe-") as td:
            dest = Path(td)
            for fx in probes:
                results.append(_try_download(app_version, app_hash, fx["assetbundleName"], dest))

        verdict = _verdict(results)
        ts = datetime.now(timezone.utc).strftime("%Y-%m-%dT%H:%M:%SZ")
        lines = [
            f"# CDN Probe — {ts}",
            "",
            f"Verdict: {verdict}",
            "",
            f"- appVersion: `{app_version}`",
            f"- appHash: `{app_hash[:8]}…` (truncated)",
            "",
            "## Probe Fixtures",
            "",
        ]
        for label, fx, r in zip(("OLDEST", "MID-LIFE", "NEWEST"), probes, results):
            lines += [
                f"### {label} — id={fx['id']} `{fx['assetbundleName']}`",
                f"- status: **{r['status']}**",
                f"- exit_code: `{r['exit_code']}`",
                f"- bytes: `{r['bytes']}`",
                f"- sha256: `{r['sha256']}`" if r["sha256"] else "- sha256: _(not downloaded)_",
                "- stderr_tail:",
                "  ```",
                "  " + (r["stderr_tail"].replace("\n", "\n  ") or "(empty)"),
                "  ```",
                "",
            ]
        if verdict == "FAILED":
            lines += [
                "## Escalation",
                "",
                "Even the oldest probe fixture failed — frozen apphash no longer authenticates against the live JP CDN.",
                "User MUST pick a mitigation before pilot can continue:",
                "- (a) Extract current apphash from a fresh APK via `sssekai apphash -s <xapk>`",
                "- (b) Accept snapshot-only pipeline (use whatever bundles are already cached locally; no future re-extraction)",
                "- (c) Find an active community fork of sekai-apphash or twintail",
                "",
            ]
        REPORT_PATH.parent.mkdir(parents=True, exist_ok=True)
        REPORT_PATH.write_text("\n".join(lines))
        print(f"Verdict: {verdict}; report -> {REPORT_PATH}")
        return 0 if verdict == "LIVE_OK" else 1

    if __name__ == "__main__":
        sys.exit(main())
    ```

    # ======== 2D 纹理抽取（D-08 平面项分支）========
    # INPUT:  Unity AssetBundle 路径
    # OUTPUT: 透明 RGBA PIL.Image
    # POS:    scripts/sprite-pipeline/pipeline/extract_2d.py

    from pathlib import Path
    from typing import Optional, Tuple
    import UnityPy
    from PIL import Image

    def extract_main_texture(bundle_path: Path) -> Image.Image:
        env = UnityPy.load(str(bundle_path))
        # 1) 优先找 Material._MainTex
        for obj in env.objects:
            if obj.type.name != "Material":
                continue
            mat = obj.read()
            tex_envs = getattr(getattr(mat, "m_SavedProperties", None), "m_TexEnvs", None)
            if not tex_envs:
                continue
            main_tex = None
            # tex_envs 在 UnityPy 里可能是 list[(name, env)] 或 dict
            try:
                items = tex_envs.items() if hasattr(tex_envs, "items") else tex_envs
            except Exception:
                items = []
            for entry in items:
                name, env_obj = entry if isinstance(entry, tuple) else (entry[0], entry[1])
                if name == "_MainTex":
                    main_tex = env_obj
                    break
            if main_tex and getattr(main_tex, "m_Texture", None):
                tex = main_tex.m_Texture.read()
                img = tex.image
                if img is not None:
                    return img.convert("RGBA")
        # 2) 回退：bundle 中第一个 Texture2D
        for obj in env.objects:
            if obj.type.name == "Texture2D":
                tex = obj.read()
                img = tex.image
                if img is not None:
                    return img.convert("RGBA")
        raise ValueError(f"no texture in bundle: {bundle_path}")

    def extract_to_png(bundle_path: Path, out_path: Path,
                      target_size: Optional[Tuple[int, int]] = None) -> Tuple[int, int]:
        img = extract_main_texture(bundle_path)
        if target_size is not None:
            img = img.resize(target_size, Image.LANCZOS)
        out_path.parent.mkdir(parents=True, exist_ok=True)
        img.save(out_path, format="PNG", optimize=True)
        return img.size
    ```

    # ======== 3-fixture 试运行（Wave 1 PILOT）========
    # INPUT:  mysekaiFixtures.json + assets-cache/bundles/<assetbundleName>.unity3d
    # OUTPUT: public/sprites/<assetbundleName>.png (× 3) + manifest.json (3 entries)
    # POS:    scripts/sprite-pipeline/pipeline/pilot.py
    # 用法：python -m pipeline.pilot [--fixtures id1,id2,id3]

    import argparse
    import json
    import os
    import shutil
    import subprocess
    import sys
    from datetime import datetime, timezone
    from pathlib import Path

    from pipeline.config import (
        BUNDLES_DIR, FIXTURES_JSON_PATH, MANIFEST_PATH,
        SPRITES_OUT_DIR, TILE_PX, PIPELINE_DIR,
    )
    from pipeline.routing import is_2d_branch
    from pipeline.extract_2d import extract_to_png

    BLENDER = os.environ.get("BLENDER_BIN", "blender")

    def _pick_default_fixtures(all_fixtures: list[dict]) -> list[dict]:
        outdoor = [f for f in all_fixtures if f.get("mysekaiSettableSiteType") != "room"]
        flat = next((f for f in outdoor if is_2d_branch(f)), None)
        small_3d = next(
            (f for f in outdoor if not is_2d_branch(f)
             and f["gridSize"]["width"] == 1 and f["gridSize"]["depth"] == 1), None)
        large_3d = next(
            (f for f in outdoor if not is_2d_branch(f)
             and (f["gridSize"]["width"] >= 3 or f["gridSize"]["depth"] >= 3)), None)
        picks = [x for x in (flat, small_3d, large_3d) if x is not None]
        if len(picks) < 3:
            raise RuntimeError(f"could not pick 3 distinct pilot fixtures (got {len(picks)})")
        return picks

    def _render_3d(fx: dict, bundle: Path, out: Path) -> tuple[bool, str]:
        # 3D 分支：Blender headless。先尝试直接喂 .unity3d（addon 路径），失败则记录给 PILOT-FINDINGS。
        result = subprocess.run([
            BLENDER, "--background",
            "--python", str(PIPELINE_DIR / "pipeline" / "blender_render.py"),
            "--",
            "--asset", str(bundle),
            "--out", str(out),
            "--tile-w", str(fx["gridSize"]["width"]),
            "--tile-d", str(fx["gridSize"]["depth"]),
        ], capture_output=True, text=True, timeout=180)
        if result.returncode == 0:
            return True, "ok"
        if result.returncode == 2 and "sssekai_blender_io unavailable" in result.stderr:
            return False, "addon-headless-failed"
        return False, f"blender exit {result.returncode}: {result.stderr[-300:]}"

    def main() -> int:
        ap = argparse.ArgumentParser()
        ap.add_argument("--fixtures", help="comma-sep fixture ids; default = auto-pick")
        args = ap.parse_args()

        with open(FIXTURES_JSON_PATH) as f:
            all_fx = json.load(f)
        if args.fixtures:
            ids = {int(x) for x in args.fixtures.split(",")}
            picks = [f for f in all_fx if f["id"] in ids]
        else:
            picks = _pick_default_fixtures(all_fx)

        SPRITES_OUT_DIR.mkdir(parents=True, exist_ok=True)
        manifest_entries: dict[str, dict] = {}
        failures = 0
        for fx in picks:
            bundle = BUNDLES_DIR / f"{fx['assetbundleName']}.unity3d"
            png = SPRITES_OUT_DIR / f"{fx['assetbundleName']}.png"
            if not bundle.exists():
                print(f"[ERR] {fx['assetbundleName']}: bundle not in cache ({bundle})", file=sys.stderr)
                failures += 1
                continue
            if is_2d_branch(fx):
                print(f"[ROUTE] {fx['assetbundleName']} -> 2D")
                try:
                    w_px = fx["gridSize"]["width"] * TILE_PX
                    d_px = fx["gridSize"]["depth"] * TILE_PX
                    size = extract_to_png(bundle, png, target_size=(w_px, d_px))
                    print(f"[OK] {fx['assetbundleName']} 2D {size}")
                    manifest_entries[fx["assetbundleName"]] = {
                        "mode": "2d",
                        "sprite": f"sprites/{fx['assetbundleName']}.png",
                        "size_px": list(size),
                    }
                except Exception as e:
                    print(f"[ERR] {fx['assetbundleName']} 2D: {e}", file=sys.stderr)
                    failures += 1
            else:
                print(f"[ROUTE] {fx['assetbundleName']} -> 3D")
                ok, info = _render_3d(fx, bundle, png)
                if ok:
                    print(f"[OK] {fx['assetbundleName']} 3D")
                    manifest_entries[fx["assetbundleName"]] = {
                        "mode": "3d",
                        "sprite": f"sprites/{fx['assetbundleName']}.png",
                        "size_px": [fx["gridSize"]["width"] * TILE_PX,
                                    fx["gridSize"]["depth"] * TILE_PX],
                    }
                else:
                    print(f"[ERR] {fx['assetbundleName']} 3D: {info}", file=sys.stderr)
                    failures += 1

        manifest = {
            "version": "1",
            "extracted_at": datetime.now(timezone.utc).strftime("%Y-%m-%dT%H:%M:%SZ"),
            "fixtures": manifest_entries,
        }
        MANIFEST_PATH.write_text(json.dumps(manifest, indent=2))
        print(f"\nPilot done: {len(manifest_entries)}/{len(picks)} OK; manifest -> {MANIFEST_PATH}")
        return 1 if failures else 0

    if __name__ == "__main__":
        sys.exit(main())
    ```
