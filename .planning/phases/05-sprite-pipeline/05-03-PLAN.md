    ```python
    # ======== sssekai abcache 包装器（D-12 配方）========
    # INPUT:  --app-version <X.Y.Z>（必填，从 sekai.best 或 apkpure 查）
    #         --app-hash <UUID>（可选；缺省自动从 mos9527/sekai-apphash 拉取）
    #         --dry-run（不联网，打印完整命令 + .extracted-with）
    # OUTPUT: scripts/sprite-pipeline/assets-cache/bundles/*.unity3d + .extracted-with
    # POS:    scripts/sprite-pipeline/pipeline/download.py

    import argparse
    import json
    import subprocess
    import sys
    import urllib.request
    from datetime import datetime, timezone
    from pathlib import Path

    from pipeline.config import ASSETS_CACHE_DIR, BUNDLES_DIR

    APPHASH_URL = "https://raw.githubusercontent.com/mos9527/sekai-apphash/master/jp/apphash.json"
    SSSEKAI_VERSION = "0.8.0"

    def fetch_current_app_hash() -> str:
        with urllib.request.urlopen(APPHASH_URL, timeout=15) as r:
            data = json.loads(r.read().decode("utf-8"))
        # apphash.json is a list of envs; pick first 'production'
        prod = next((d for d in data if "production" in d.get("environment", "")), None)
        if not prod:
            raise RuntimeError(f"no production entry in {APPHASH_URL}")
        return prod["appHash"]

    def build_command(app_version: str, app_hash: str) -> list[str]:
        return [
            "sssekai", "abcache",
            "--db", str(ASSETS_CACHE_DIR / "abcache.db"),
            "--app-region", "jp",
            "--app-version", app_version,
            "--app-appHash", app_hash,
            "--download-filter", ".*mysekai.*",
            "--download-ensure-deps",
            "--download-dir", str(BUNDLES_DIR),
        ]

    def write_extracted_with(app_version: str, app_hash: str) -> None:
        ASSETS_CACHE_DIR.mkdir(parents=True, exist_ok=True)
        body = (
            f"sssekai_version: {SSSEKAI_VERSION}\n"
            f"app_version: {app_version}\n"
            f"app_hash: {app_hash}\n"
            f"extracted_at: {datetime.now(timezone.utc).strftime('%Y-%m-%dT%H:%M:%SZ')}\n"
        )
        (ASSETS_CACHE_DIR / ".extracted-with").write_text(body)

    def add_args(p: argparse.ArgumentParser) -> None:
        p.add_argument("--app-version", required=False,
                       help="JP app version (e.g. 6.4.1). Required for non-dry-run.")
        p.add_argument("--app-hash", help="Override; default = fetch from mos9527/sekai-apphash")
        p.add_argument("--dry-run", action="store_true")

    def run(args: argparse.Namespace) -> int:
        if not args.app_version:
            if not args.dry_run:
                print("--app-version required for non-dry-run", file=sys.stderr)
                return 2
            args.app_version = "<APP_VERSION>"
        app_hash = args.app_hash or ("<APP_HASH>" if args.dry_run else fetch_current_app_hash())
        cmd = build_command(args.app_version, app_hash)
        if args.dry_run:
            print(" ".join(cmd))
            print(f"\n# .extracted-with would contain:")
            print(f"sssekai_version: {SSSEKAI_VERSION}")
            print(f"app_version: {args.app_version}")
            print(f"app_hash: {app_hash}")
            return 0
        BUNDLES_DIR.mkdir(parents=True, exist_ok=True)
        result = subprocess.run(cmd, check=False)
        if result.returncode != 0:
            return result.returncode
        write_extracted_with(args.app_version, app_hash)
        return 0
    ```

    ```python
    # ======== 3D 批渲染调度 ========
    # INPUT:  outdoor 3D-branch 装备列表 + assets-cache/bundles/
    # OUTPUT: public/sprites/<assetbundleName>.png
    # POS:    scripts/sprite-pipeline/pipeline/render_3d.py

    import argparse
    import json
    import os
    import subprocess
    import sys
    from pathlib import Path
    from tqdm import tqdm

    from pipeline.config import (BUNDLES_DIR, FIXTURES_JSON_PATH, PIPELINE_DIR,
                                  SPRITES_OUT_DIR, TILE_PX)
    from pipeline.routing import is_2d_branch, is_outdoor

    BLENDER = os.environ.get("BLENDER_BIN", "blender")
    # Path A or B 由 PILOT-FINDINGS 决定；这里假设 Path A（如改为 B，引入 glb_writer）。
    USE_GLB_FALLBACK = False  # ← 由 PLAN 03 Task 2 作者根据 PILOT-FINDINGS 设置

    def _input_for_blender(bundle: Path, tmp_dir: Path) -> Path:
        if not USE_GLB_FALLBACK:
            return bundle
        from pipeline.glb_writer import unity_bundle_to_glb
        glb = tmp_dir / f"{bundle.stem}.glb"
        unity_bundle_to_glb(bundle, glb)
        return glb

    def _outdoor_3d(fixtures: list[dict]) -> list[dict]:
        return [f for f in fixtures if is_outdoor(f) and not is_2d_branch(f)]

    def _render_one(fx: dict, tmp_dir: Path) -> tuple[str, bool, str]:
        bundle = BUNDLES_DIR / f"{fx['assetbundleName']}.unity3d"
        out = SPRITES_OUT_DIR / f"{fx['assetbundleName']}.png"
        if not bundle.exists():
            return fx["assetbundleName"], False, "no-bundle"
        try:
            asset = _input_for_blender(bundle, tmp_dir)
        except Exception as e:
            return fx["assetbundleName"], False, f"glb-convert: {e}"
        r = subprocess.run([
            BLENDER, "--background",
            "--python", str(PIPELINE_DIR / "pipeline" / "blender_render.py"),
            "--",
            "--asset", str(asset),
            "--out", str(out),
            "--tile-w", str(fx["gridSize"]["width"]),
            "--tile-d", str(fx["gridSize"]["depth"]),
        ], capture_output=True, text=True, timeout=180)
        return fx["assetbundleName"], r.returncode == 0, (r.stderr[-200:] if r.returncode else "ok")

    def add_args(p: argparse.ArgumentParser) -> None:
        p.add_argument("--dry-run", action="store_true")
        p.add_argument("--limit", type=int, default=None, help="render only first N fixtures (debugging)")

    def run(args: argparse.Namespace) -> int:
        with open(FIXTURES_JSON_PATH) as f:
            all_fx = json.load(f)
        targets = _outdoor_3d(all_fx)
        if args.limit:
            targets = targets[: args.limit]
        if args.dry_run:
            print(f"render-3d dry-run: {len(targets)} fixtures")
            return 0
        SPRITES_OUT_DIR.mkdir(parents=True, exist_ok=True)
        import tempfile
        with tempfile.TemporaryDirectory() as tmp:
            tmp_dir = Path(tmp)
            failures = 0
            for fx in tqdm(targets, desc="render 3D"):
                _, ok, info = _render_one(fx, tmp_dir)
                if not ok:
                    failures += 1
                    print(f"[ERR] {fx['assetbundleName']}: {info}", file=sys.stderr)
        print(f"render-3d done: {len(targets) - failures}/{len(targets)} OK")
        return 1 if failures and failures > len(targets) // 2 else 0
    ```

    ```python
    # ======== 清单组装（SPRT-05/06）========
    # INPUT:  public/sprites/*.png + mysekaiFixtures.json
    # OUTPUT: public/sprites/manifest.json
    # POS:    scripts/sprite-pipeline/pipeline/assemble_manifest.py

    import argparse
    import json
    import sys
    from datetime import datetime, timezone

    from pipeline.config import (FIXTURES_JSON_PATH, MANIFEST_PATH, SPRITES_OUT_DIR, TILE_PX)
    from pipeline.routing import is_2d_branch, is_outdoor

    def add_args(p: argparse.ArgumentParser) -> None:
        p.add_argument("--dry-run", action="store_true")
        p.add_argument("--check-size", action="store_true",
                       help="post-write 150MB cap check; exits 1 if exceeded")

    def run(args: argparse.Namespace) -> int:
        with open(FIXTURES_JSON_PATH) as f:
            all_fx = json.load(f)
        outdoor = [f for f in all_fx if is_outdoor(f)]
        # uniqueness check
        seen: dict[str, int] = {}
        kept: list[dict] = []
        for fx in outdoor:
            ab = fx["assetbundleName"]
            if ab in seen:
                print(f"[DUP] {ab} (id {fx['id']} duplicates id {seen[ab]}); keeping first",
                      file=sys.stderr)
                continue
            seen[ab] = fx["id"]
            kept.append(fx)
        entries: dict[str, dict] = {}
        for fx in kept:
            png = SPRITES_OUT_DIR / f"{fx['assetbundleName']}.png"
            if not png.exists():
                continue  # not all fixtures will have PNGs (bundle missing OR render error)
            mode = "2d" if is_2d_branch(fx) else "3d"
            entries[fx["assetbundleName"]] = {
                "mode": mode,
                "sprite": f"sprites/{fx['assetbundleName']}.png",
                "size_px": [fx["gridSize"]["width"] * TILE_PX,
                            fx["gridSize"]["depth"] * TILE_PX],
            }
        manifest = {
            "version": "1",
            "extracted_at": datetime.now(timezone.utc).strftime("%Y-%m-%dT%H:%M:%SZ"),
            "fixtures": entries,
        }
        if args.dry_run:
            print(f"assemble-manifest dry-run: {len(entries)} entries")
            return 0
        SPRITES_OUT_DIR.mkdir(parents=True, exist_ok=True)
        MANIFEST_PATH.write_text(json.dumps(manifest, indent=2))
        print(f"manifest written: {len(entries)} entries -> {MANIFEST_PATH}")
        if args.check_size:
            from pipeline.check_size import check_size
            return check_size()
        return 0
    ```

    ```python
    # ======== public/sprites 体积闸（D-02 150MB 上限）========
    # POS:    scripts/sprite-pipeline/pipeline/check_size.py
    import sys
    from pipeline.config import SPRITES_OUT_DIR, SIZE_BUDGET_MB

    def check_size() -> int:
        total = sum(p.stat().st_size for p in SPRITES_OUT_DIR.rglob("*") if p.is_file())
        mb = total / (1024 * 1024)
        if mb > SIZE_BUDGET_MB:
            print(f"[FAIL] public/sprites/ = {mb:.1f} MB > {SIZE_BUDGET_MB} MB cap (D-02)", file=sys.stderr)
            return 1
        print(f"[OK] public/sprites/ = {mb:.1f} MB (cap {SIZE_BUDGET_MB} MB)")
        return 0

    if __name__ == "__main__":
        sys.exit(check_size())
    ```
