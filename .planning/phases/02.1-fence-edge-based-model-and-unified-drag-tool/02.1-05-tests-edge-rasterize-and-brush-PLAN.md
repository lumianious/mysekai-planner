      })

        expect(dominantAxis(3, 3)).toBe('h')
        expect(dominantAxis(-3, 3)).toBe('h')
        expect(dominantAxis(0, 0)).toBe('h')
      })
    })

      it('horizontal (0,0)→(3,0) yields 3 horizontal edges at x=0,1,2', () => {

        ).toEqual([
          { fixtureId: 1, x: 0, y: 0, orientation: 'h' },
          { fixtureId: 1, x: 1, y: 0, orientation: 'h' },

        ])
