    2. Create `src/hooks/useCanvasInteraction.ts`:

    Custom hook managing pan/zoom state for the Konva Stage:
    ```typescript
    export function useCanvasInteraction() {
      const [stagePos, setStagePos] = useState({ x: 0, y: 0 });
      const [stageScale, setStageScale] = useState(1);

      const handleWheel = useCallback((e: KonvaEventObject<WheelEvent>) => {
        // Pattern 3 from RESEARCH.md:
        // Zoom relative to pointer position
        // SCALE_BY = 1.05, MIN_SCALE = 0.15, MAX_SCALE = 3.0
        // MIN_SCALE = 0.15 to fit 100x100 grid (100*32=3200px at 0.15 = 480px viewport)
        // MAX_SCALE = 3.0 for detail inspection
        e.evt.preventDefault();
        const stage = e.target.getStage();
        if (!stage) return;
        const oldScale = stage.scaleX();
        const pointer = stage.getPointerPosition();
        if (!pointer) return;
        const mousePointTo = {
          x: (pointer.x - stage.x()) / oldScale,
          y: (pointer.y - stage.y()) / oldScale,
        };
        const direction = e.evt.deltaY > 0 ? -1 : 1;
        const SCALE_BY = 1.05;
        const newScale = Math.min(3.0, Math.max(0.15,
          direction > 0 ? oldScale * SCALE_BY : oldScale / SCALE_BY
        ));
        setStageScale(newScale);
        setStagePos({
          x: pointer.x - mousePointTo.x * newScale,
          y: pointer.y - mousePointTo.y * newScale,
        });
      }, []);

      return { stagePos, setStagePos, stageScale, handleWheel };
    }
    ```

    const [mouseGridPos, setMouseGridPos] = useState<{x: number, y: number} | null>(null);
