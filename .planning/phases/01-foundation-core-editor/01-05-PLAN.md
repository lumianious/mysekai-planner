    2. Create `src/hooks/useCanvasInteraction.ts`:

    Custom hook managing pan/zoom state for the Konva Stage:
    ```typescript
    export function useCanvasInteraction() {
      const [stagePos, setStagePos] = useState({ x: 0, y: 0 });
      const [stageScale, setStageScale] = useState(1);

      const handleWheel = useCallback((e: KonvaEventObject<WheelEvent>) => {
        // Pattern 3 from RESEARCH.md:
        // Zoom relative to pointer position
        // SCALE_BY = 1.05, MIN_SCALE = 0.15, MAX_SCALE = 3.0
        // MIN_SCALE = 0.15 to fit 100x100 grid (100*32=3200px at 0.15 = 480px viewport)
        // MAX_SCALE = 3.0 for detail inspection
        e.evt.preventDefault();
        const stage = e.target.getStage();
        if (!stage) return;
        const oldScale = stage.scaleX();
        const pointer = stage.getPointerPosition();
        if (!pointer) return;
        const mousePointTo = {
          x: (pointer.x - stage.x()) / oldScale,
          y: (pointer.y - stage.y()) / oldScale,
        };
        const direction = e.evt.deltaY > 0 ? -1 : 1;
        const SCALE_BY = 1.05;
        const newScale = Math.min(3.0, Math.max(0.15,
          direction > 0 ? oldScale * SCALE_BY : oldScale / SCALE_BY
        ));
        setStageScale(newScale);
        setStagePos({
          x: pointer.x - mousePointTo.x * newScale,
          y: pointer.y - mousePointTo.y * newScale,
        });
      }, []);

      return { stagePos, setStagePos, stageScale, handleWheel };
    }
    ```

    1. Create `src/utils/ghostPreview.ts`:

    Extract the ghost preview validity logic into a pure, testable function:
    ```typescript
    import type { PlacedItem, Fixture, Rotation, ItemLayer, GridSize } from '../types/editor';
    import { getEffectiveSize } from './grid';
    import { buildOccupancyGrid, checkCanPlace } from '../stores/editorStore';

    // ======== 放置预览有效性检查 ========

    export interface GhostValidityResult {
      valid: boolean;
      effectiveWidth: number;
      effectiveDepth: number;
    }

    export function checkGhostValidity(
      gridX: number,
      gridY: number,
      fixture: Fixture,
      rotation: Rotation,
      layer: ItemLayer,
      placedItems: Record<string, PlacedItem>,
      fixtureMap: Map<number, Fixture>,
      gridSize: GridSize,
      overwriteEnabled: boolean
    ): GhostValidityResult {
      const [w, d] = getEffectiveSize(fixture.gridSize, rotation);
      if (overwriteEnabled) {
        // 覆盖模式下只检查边界
        const inBounds = gridX >= 0 && gridY >= 0 && gridX + w <= gridSize.width && gridY + d <= gridSize.depth;
        return { valid: inBounds, effectiveWidth: w, effectiveDepth: d };
      }
      const occupancy = buildOccupancyGrid(placedItems, fixtureMap, layer);
      const canPlace = checkCanPlace(occupancy, gridX, gridY, w, d, gridSize.width, gridSize.depth);
      return { valid: canPlace, effectiveWidth: w, effectiveDepth: d };
    }
    ```
