    **Safari-safe focus (Pitfall 1 from RESEARCH.md):** Use `tabIndex={-1}` (not `tabIndex={1}`) on the container. Call `.focus()` on container click. Set `container.style.outline = 'none'`.

    Key mappings per UI-SPEC Keyboard Shortcuts:

    ```typescript
    const handleKeyDown = useCallback((e: KeyboardEvent) => {
      const state = useEditorStore.getState();
      const { undo, redo } = useEditorStore.temporal.getState();

      // 全局快捷键
      switch (e.key.toLowerCase()) {
        case 'v': state.setToolMode('select'); break;
        case 'b': state.setToolMode('stamp'); break;
        case 'x': state.setToolMode('remove'); break;
        case 'escape':
          state.setSelectedItem(null);
          if (state.toolMode === 'stamp') state.setActiveFixture(null);
          break;
        case 'z':
          if (e.ctrlKey || e.metaKey) {
            e.preventDefault();
            if (e.shiftKey) redo(); else undo();
          }
          break;
        case 'y':
          if (e.ctrlKey || e.metaKey) { e.preventDefault(); redo(); }
          break;
      }

      // 旋转（选中项目或预览）
      if (e.key.toLowerCase() === 'r') {
        e.preventDefault();
        const direction = e.shiftKey ? 'ccw' : 'cw';
        if (state.selectedItemId) {
          state.rotateItem(state.selectedItemId, direction);
        }
        // For preview rotation in stamp mode, update the local previewRotation
        // via a callback passed to the hook
        onRotatePreview?.(direction);
      }

      // 热栏数字键
      if (e.key >= '1' && e.key <= '9' && !isHoveringCatalog) {
        state.activateHotbar(parseInt(e.key));
      }

      // 删除选中项
      if ((e.key === 'Delete' || e.key === 'Backspace') && state.selectedItemId) {
        e.preventDefault();
        state.removeItem(state.selectedItemId);
      }

      // 方向键微调（选择模式下移动选中项目）
      if (['ArrowUp','ArrowDown','ArrowLeft','ArrowRight'].includes(e.key) && state.selectedItemId) {
        e.preventDefault();
        const item = state.placedItems[state.selectedItemId];
        if (!item) return;
        const dx = e.key === 'ArrowLeft' ? -1 : e.key === 'ArrowRight' ? 1 : 0;
        const dy = e.key === 'ArrowUp' ? -1 : e.key === 'ArrowDown' ? 1 : 0;
        // Call moveItem with validation (via useEditorActions)
        onNudge?.(state.selectedItemId, item.x + dx, item.y + dy);
      }

      // Tab循环选择同一格子上的重叠物品
      if (e.key === 'Tab') {
        e.preventDefault();
        onCycleSelection?.();
      }
    }, [isHoveringCatalog]);
    ```

    Hook parameters: `{ containerRef, onRotatePreview, onNudge, onCycleSelection, isHoveringCatalog }`

    The `isHoveringCatalog` flag prevents number keys from activating hotbar while hovering catalog items (D-08 assignment takes priority).

    2. Create `src/hooks/useEditorActions.ts`:

    Higher-level action hook that combines store actions with validation logic:

    ```typescript
    export function useEditorActions(fixtureMap: Map<number, Fixture>) {
      const store = useEditorStore;

      const handleCanvasClick = useCallback((gridX: number, gridY: number) => {
        const state = store.getState();
        const { toolMode, activeFixtureId, overwriteEnabled } = state;

        if (toolMode === 'stamp' && activeFixtureId !== null) {
          const fixture = fixtureMap.get(activeFixtureId);
          if (!fixture) return;
          const layer = getItemLayer(fixture);
          const [w, d] = getEffectiveSize(fixture.gridSize, previewRotation);
          const occupancy = buildOccupancyGrid(state.placedItems, fixtureMap, layer);

          if (overwriteEnabled) {
            // Remove existing items in the footprint before placing
            removeItemsInFootprint(gridX, gridY, w, d, occupancy, state);
          }

          if (!checkCanPlace(occupancy, gridX, gridY, w, d, state.gridSize.width, state.gridSize.depth)) {
            return; // Blocked — ghost was red
          }

          state.placeItem({
            fixtureId: activeFixtureId,
            x: gridX, y: gridY,
            rotation: previewRotation,
            layer,
            isSystem: false,
          });
          // Stamp mode persists (D-06) — don't reset activeFixtureId
        }

        if (toolMode === 'remove') {
          // Find item at clicked grid position
          const itemAtPos = findItemAtPosition(state.placedItems, fixtureMap, gridX, gridY);
          if (itemAtPos) state.removeItem(itemAtPos.id);
        }

        if (toolMode === 'select') {
          // Find item at clicked grid position for selection
          const itemAtPos = findItemAtPosition(state.placedItems, fixtureMap, gridX, gridY);
          state.setSelectedItem(itemAtPos?.id ?? null);
        }
      }, [fixtureMap, previewRotation]);

      const handleMoveItem = useCallback((itemId: string, newX: number, newY: number) => {
        const state = store.getState();
        const item = state.placedItems[itemId];
        if (!item) return;
        const fixture = fixtureMap.get(item.fixtureId);
        if (!fixture) return;
        const [w, d] = getEffectiveSize(fixture.gridSize, item.rotation);
        const occupancy = buildOccupancyGrid(state.placedItems, fixtureMap, item.layer);
        if (checkCanPlace(occupancy, newX, newY, w, d, state.gridSize.width, state.gridSize.depth, itemId)) {
          state.moveItem(itemId, newX, newY);
        }
      }, [fixtureMap]);

      return { handleCanvasClick, handleMoveItem };
    }
    ```

    Helper: `findItemAtPosition(placedItems, fixtureMap, gridX, gridY)` — iterate placed items, compute footprint, check if gridX/gridY falls within any item's footprint. Return the item or null. For overlapping items on different layers, prioritize furniture layer.

    Helper: `removeItemsInFootprint(x, y, w, d, occupancy, state)` — for overwrite mode, find and remove all items whose tiles overlap the stamp footprint.

    const undoWithFlash = () => {
      const beforeItems = useEditorStore.getState().placedItems;
      useEditorStore.temporal.getState().undo();
