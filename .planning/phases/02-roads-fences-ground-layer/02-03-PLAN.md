    ```typescript
    // ======== 网格线栅格化 ========
    // INPUT: 起止格子坐标 + 可选步长（默认 1，画刷传 fixtureWidth）
    // OUTPUT: 从 start 到 end 顺序排列的所有中间格子坐标数组
    // POS: src/utils/rasterize.ts — 纯函数，供 EditorCanvas 拖拽画刷与围栏线工具使用
    //
    // 使用 4 连通 Bresenham（保证相邻样本共享一条边）。
    // 对画刷：step 取 fixture.gridSize.width（v1 全部为 2），保证相邻样本不重叠。
    // 对围栏线：调用方先用 snapToAxis 对齐端点，Bresenham 退化为直线迭代。
    //
    // 算法细节：先将输入映射到 step-space (除以 step)，走 Bresenham，再乘回 step
    // 输出 — 这样每个输出坐标都是 step 的整数倍，自动对齐。

    export interface Tile {
      x: number
      y: number
    }

    // ======== Bresenham 4 连通栅格化 ========

    export function rasterizeLine(
      x0: number,
      y0: number,
      x1: number,
      y1: number,
      step = 1,
    ): Tile[] {
      if (step <= 0) throw new Error('rasterizeLine: step must be > 0')

      // 映射到 step-space
      let sx = Math.round(x0 / step)
      let sy = Math.round(y0 / step)
      const ex = Math.round(x1 / step)
      const ey = Math.round(y1 / step)

      const tiles: Tile[] = []
      const dx = Math.abs(ex - sx)
      const dy = -Math.abs(ey - sy)
      const stepX = sx < ex ? 1 : -1
      const stepY = sy < ey ? 1 : -1
      let err = dx + dy

      // 安全限制：避免异常输入导致死循环
      const maxIter = dx + Math.abs(dy) + 2
      let iter = 0

      while (iter++ < maxIter + 1) {
        tiles.push({ x: sx * step, y: sy * step })
        if (sx === ex && sy === ey) break
        const e2 = 2 * err
        if (e2 >= dy) { err += dy; sx += stepX }
        if (e2 <= dx) { err += dx; sy += stepY }
      }

      return tiles
    }

    // ======== 轴对齐吸附（围栏线工具 D-35） ========
    // 将自由端点强制为同一行或同一列（dx==dy 时默认水平）

    export function snapToAxis(
      startX: number,
      startY: number,
      endX: number,
      endY: number,
    ): Tile {
      const dx = Math.abs(endX - startX)
      const dy = Math.abs(endY - startY)
      return dx >= dy
        ? { x: endX, y: startY } // 水平线（包括 dx==dy 的平局）
        : { x: startX, y: endY } // 垂直线
    }
    ```

  <behavior>
    **Drag-paint behavior (ROAD-01, D-31, D-33, D-40, D-41):**
    - In brush mode with a drag-paint fixture active, pressing mouse down on an empty ground tile places one fixture at that snapped tile and calls `startStrokeBatch`
    - Dragging the mouse while down causes every 4-connected tile from the last painted tile to the current snapped tile to receive a placement (via rasterizeLine with step = fixture.gridSize.width)
    - Releasing the mouse calls `endStrokeBatch`, producing exactly 1 zundo history entry for the full stroke
    - With overwrite=OFF: painting a tile that already has a ground-layer item is SKIPPED but does NOT abort the stroke (RESEARCH open question 3)
    - With overwrite=ON: painting a tile that already has a ground-layer item REMOVES the old item first, then places the new one
    - Painting a ground tile at (x,y) does NOT prevent placing furniture at (x,y) — the two layers are independent (D-41)
    - Fixtures are snapped to step-space: drag from editor tile (5,5) to (5,8) with a 2x2 road places fixtures at (4,4),(4,6),(4,8) (after rounding to nearest even multiple, which matches the fixture footprint alignment)

    **Drag-erase behavior (D-42):**
    - In remove mode, pressing mouse down and dragging across ground items removes every ground item touched, batched as a single undo step
    - Furniture items under the drag are NOT removed (Phase 1 D-21 — furniture still requires single click, which the existing `handleStageClick` handles)
    - The drag-erase uses the same rasterizer with step=1 (erase granularity is one editor tile, not one fixture footprint)

    **Cleanup (R-06):**
    - If the user drags off the canvas (onMouseLeave) while a stroke is active, `endStrokeBatch` is called immediately and `isPaintingRef.current` resets
    - If `window.blur` fires while a stroke is active, same cleanup runs
    - After cleanup, zundo is no longer in paused state (`temporal.isTracking === true`)

    **Ghost preview (brush mode):**
    - In brush mode with a drag-paint fixture active, the ghost preview renders a single-fixture footprint at the cursor position, colored green when valid / red when blocked (same semantics as stamp ghost)
    - In brush mode with a fence fixture active, the ghost preview can be absent (fence line tool ghost will be added in plan 02-04)

    ```typescript
    const pointerToGrid = useCallback(
      (stage: Konva.Stage, step: number) => {
        const pointer = stage.getPointerPosition()
        if (!pointer) return null
        const scale = stage.scaleX()
        // 原始格子坐标
        const rawGridX = Math.floor((pointer.x - stage.x()) / scale / TILE_SIZE)
        const rawGridY = Math.floor((pointer.y - stage.y()) / scale / TILE_SIZE)
        // 按 step 对齐（Math.floor 将 rawGridX 向下取整到 step 的倍数）
        const snappedX = Math.floor(rawGridX / step) * step
        const snappedY = Math.floor(rawGridY / step) * step
        return { gridX: snappedX, gridY: snappedY, rawGridX, rawGridY }
      },
      [],
    )
    ```
