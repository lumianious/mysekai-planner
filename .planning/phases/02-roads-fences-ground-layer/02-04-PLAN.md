  <behavior>
    **State machine:**
    - `idle`: no fence line in progress; user can click to set start
    - `picking-end`: start tile is set; ghost preview follows cursor to current tile (axis-snapped); click sets end
    - `confirming`: end tile is set; overlay shown; Enter/button confirms, Escape/button cancels, click elsewhere cancels+starts new line at clicked tile
    - Transitioning to a non-brush tool, or deactivating the fixture, resets to `idle`

    **Ghost preview:**
    - In `picking-end` phase, compute `snapped = snapToAxis(start, currentMouse)` and render all rasterized tiles between start and snapped as semi-transparent fence rectangles (same color scheme as GhostPreview — green if valid, red if any collision)
    - In `confirming` phase, render the same preview plus the overlay

    **Commit:**
    - On confirm: wrap `placeItem` for every rasterized tile in `withBatchedUndo(() => {...})`
    - Skip tiles that collide (overwrite semantics follow stamp: OFF blocks, ON removes + replaces). For v1, keep it simple: if overwrite=OFF and any tile collides, still place the non-colliding tiles and log a console.debug; if overwrite=ON, remove colliding ground items first per tile then place.
    - After commit, reset state to `idle`

    **Axis snap (D-35):**
    - `snapToAxis` already tie-breaks horizontal; use it as-is
    - Rasterization step = fence.gridSize.width (2 for all v1 fences)

    **Single-undo guarantee (D-43):**
    - A 3-tile fence line must produce exactly 1 zundo history entry

    **Keyboard:**
    - Enter confirms in `confirming` phase (no-op in other phases)
    - Escape cancels in `picking-end` or `confirming` phases (returns to idle, preserves active fixture)
  </behavior>

    **Step A — Create `src/components/canvas/FenceLineTool.tsx`:**

    ```tsx
    // ======== 围栏线工具鬼影 ========
    // INPUT: start / end 格子坐标，fixture（用于 gridSize 与图层）
    // OUTPUT: Konva Layer 渲染 rasterizeLine 产生的每个 tile 作为半透明矩形
    // POS: src/components/canvas/FenceLineTool.tsx — 围栏线预览图层

    import { Layer, Rect } from 'react-konva'
    import { rasterizeLine } from '../../utils/rasterize'
    import { TILE_SIZE } from '../../utils/grid'
    import { buildOccupancyGrid, checkCanPlace } from '../../stores/editorStore'
    import { useEditorStore } from '../../stores/editorStore'
    import type { Fixture } from '../../types/editor'

    const VALID_COLOR = '#22c55e'
    const BLOCKED_COLOR = '#ef4444'
    const GHOST_OPACITY = 0.4

    interface FenceLineToolProps {
      start: { x: number; y: number } | null
      end: { x: number; y: number } | null
      fixture: Fixture
      fixtureMap: Map<number, Fixture>
    }

    export function FenceLineTool({ start, end, fixture, fixtureMap }: FenceLineToolProps) {
      const placedItems = useEditorStore((s) => s.placedItems)
      const gridSize = useEditorStore((s) => s.gridSize)

      if (!start || !end) return <Layer listening={false} />

      const step = fixture.gridSize.width
      const tiles = rasterizeLine(start.x, start.y, end.x, end.y, step)
      const groundOcc = buildOccupancyGrid(placedItems, fixtureMap, 'ground')

      return (
        <Layer listening={false}>
          {tiles.map((t, i) => {
            const canPlace = checkCanPlace(
              groundOcc, t.x, t.y, step, fixture.gridSize.depth,
              gridSize.width, gridSize.depth,
            )
            const color = canPlace ? VALID_COLOR : BLOCKED_COLOR
            return (
              <Rect
                key={`${t.x},${t.y},${i}`}
                x={t.x * TILE_SIZE}
                y={t.y * TILE_SIZE}
                width={step * TILE_SIZE}
                height={fixture.gridSize.depth * TILE_SIZE}
                fill={color}
                stroke={color}
                strokeWidth={1}
                opacity={GHOST_OPACITY}
                cornerRadius={2}
              />
            )
          })}
        </Layer>
      )
    }
    ```

    ```typescript
    const confirmFenceLine = useCallback(() => {
      if (fenceState.phase !== 'confirming') return
      if (!fenceState.start || !fenceState.end) return
      const fixture = activeFenceFixture
      if (!fixture) return
      const step = fixture.gridSize.width
      const depth = fixture.gridSize.depth
      const tiles = rasterizeLine(
        fenceState.start.x, fenceState.start.y,
        fenceState.end.x, fenceState.end.y, step,
      )
      withBatchedUndo(() => {
        const state = useEditorStore.getState()
        const overwrite = state.overwriteEnabled
        for (const t of tiles) {
          const occ = buildOccupancyGrid(
            useEditorStore.getState().placedItems, fixtureMap, 'ground',
          )
          const can = checkCanPlace(
            occ, t.x, t.y, step, depth, state.gridSize.width, state.gridSize.depth,
          )
          if (can) {
            state.placeItem({
              fixtureId: fixture.id,
              x: t.x, y: t.y, rotation: 0, layer: 'ground', isSystem: false,
            })
          } else if (overwrite) {
            // 移除该 tile 占位 ground 项
            for (let dx = 0; dx < step; dx++) {
              for (let dy = 0; dy < depth; dy++) {
                const key = `${t.x + dx},${t.y + dy}`
                const occupantId = occ.get(key)
                if (occupantId) state.removeItem(occupantId)
              }
            }
            const fresh = buildOccupancyGrid(
              useEditorStore.getState().placedItems, fixtureMap, 'ground',
            )
            if (checkCanPlace(fresh, t.x, t.y, step, depth, state.gridSize.width, state.gridSize.depth)) {
              state.placeItem({
                fixtureId: fixture.id,
                x: t.x, y: t.y, rotation: 0, layer: 'ground', isSystem: false,
              })
            }
          }
          // overwrite=OFF + 被占 → 跳过（与 brush 拖拽一致）
        }
      })
      setFenceState({ phase: 'idle', start: null, end: null })
    }, [fenceState, activeFenceFixture, fixtureMap])

    const cancelFenceLine = useCallback(() => {
      setFenceState({ phase: 'idle', start: null, end: null })
    }, [])
    ```
